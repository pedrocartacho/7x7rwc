// Battleground Engine
// *********************************************************************

-	script	kvm52_BG	-1,{
	end;

OnGuillaumeJoin:
OnCroixJoin:
	donpcevent "kvm52_BG::OnReadyCheck";
	end;

OnGuillaumeQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnGuillaumeDie:
	if( $@kvm52BG == 2 )
	{
		set .Guillaume_Count, .Guillaume_Count - 1;
		set .Croix_Score, .Croix_Score + 1;
		bg_updatescore "1@pump",.Guillaume_Score,.Croix_Score;
		if( .Guillaume_Count < 1 ){
			donpcevent "kvm52_BG::OnCroixWin";
		}
	}
	end;

OnCroixQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnCroixDie:
	if( $@kvm52BG == 2 )
	{
		set .Croix_Count, .Croix_Count - 1;
		set .Guillaume_Score, .Guillaume_Score + 1;
		bg_updatescore "1@pump",.Guillaume_Score,.Croix_Score;
		if( .Croix_Count < 1 ){
			donpcevent "kvm52_BG::OnGuillaumeWin";
		}
	}
	end;

OnReadyCheck:
	if( $@kvm52BG )
		end;
	set .@Guillaume, getwaitingroomstate(0,"kvm52R_Guillaume");
	set .@Croix, getwaitingroomstate(0,"kvm52R_Croix");

	if((.@Guillaume < $@jogadoresblue) || (.@Croix < $@jogadoresred))
	{
		if( .@Guillaume > 1 && .@Croix > 1 && !agitcheck() && $@KvM_Flood < gettimetick(2) )
		{
			announce "Arena 22 - 1@pump | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",0,0x00bfff;
			set $@KvM_Flood, gettimetick(2) + 15;
		}
		else
			mapannounce "quiz_02","Arena 22 - 1@pump | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",1,0x00bfff;
		end;
	}

	if ((.@Guillaume == 0) || (.@Croix == 0)) {
		end;
	}
	
	set $@kvm52BG, 1; // Starting
	donpcevent "kvm52R_Croix::OnEnterBG";
	donpcevent "kvm52R_Guillaume::OnEnterBG";
	donpcevent "kvm52_BG::OnStart";
	end;

OnStart:
	announce "- Duas equipes entraram na [ Arena 22 - 1@pump ]. -", bc_yellow;
	set $@kvm52BG_Victory, 0;
	// Warp Teams
	bg_warp $@kvm52BG_id1,"1@pump",36,128;
	bg_warp $@kvm52BG_id2,"1@pump",83,76;

	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }
	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }

	setwall "1@pump",30,124,11,6,0,"batc20wall_a";
	setwall "1@pump",40,125,3,0,0,"batc20wall_b";
	setwall "1@pump",79,80,8,6,0,"batc20wall_c";
	setwall "1@pump",79,79,8,4,0,"batc20wall_d";

	sleep 100;
	// ScoreBoard
	
	set .Guillaume_Score, 0;
	set .Croix_Score, 0;
	initnpctimer;
	end;

OnTimer3000:
	mapannounce "1@pump","- The match will begin in 15 seconds! -", bc_yellow;
	end;

OnTimer15000:
	mapannounce "1@pump","- The match will begin in 5 seconds! -", bc_yellow;
	end;

OnTimer16000:
	mapannounce "1@pump","- The match will begin in 4 seconds! -", bc_yellow;
	end;

OnTimer17000:
	mapannounce "1@pump","- The match will begin in 3 seconds! -", bc_yellow;
	end;

OnTimer18000:
	mapannounce "1@pump","- The match will begin in 2 seconds! -", bc_yellow;
	end;
	
OnTimer19000:
	mapannounce "1@pump","- The match will begin in 1 second! -", bc_yellow;
	end;

OnTimer20000:
	// Team Members
	set .Guillaume_Count, bg_get_data($@kvm52BG_id1, 0);
	set .Croix_Count, bg_get_data($@kvm52BG_id2, 0);
	if( (.Guillaume_Count < $@jogadoresblue) || (.Croix_Count < $@jogadoresred) )
	{
		set $@kvm52BG_Victory, 3;
		set $@kvm52BG, 3;
		mapannounce "1@pump","- There are not enough players to start the match. -", bc_yellow;
		stopnpctimer;
		sleep 2000;
		donpcevent "kvm52_BG::OnStop";
		end;
	}

	set $@kvm52BG, 2; // Playing
	delwall "batc20wall_a";
	delwall "batc20wall_b";
	delwall "batc20wall_c";
	delwall "batc20wall_d";
	end;

OnTimer25001:
	mapannounce "1@pump","- The match started! Good luck to all. -", bc_yellow;
killmonsterall "1@pump";
	end;

OnTimer420000:
	mapannounce "1@pump","- The match will end in 30 seconds! -", bc_yellow;
	end;

OnTimer450000:
	if( .Croix_Count > .Guillaume_Count )
		donpcevent "kvm52_BG::OnCroixWin";
	else if( .Croix_Count < .Guillaume_Count )
		donpcevent "kvm52_BG::OnGuillaumeWin";
	else
	{ // Draw Game
		set $@kvm52BG, 3;
		set $@kvm52BG_Victory, 3;

		stopnpctimer;
		sleep 2000;
		announce "- As duas equipes empataram! -", bc_yellow;
		donpcevent "kvm52_BG::OnStop";
	}
	end;

OnGuillaumeWin:
	set $@kvm52BG, 3;
	set $@kvm52BG_Victory, 1;

	stopnpctimer;
	sleep 1000;	
	announce "- Blue Team VENCEU! -",bc_all,0x00bfff;
	sleep 3000;
	donpcevent "kvm52_BG::OnStop";
	end;

OnCroixWin:
	set $@kvm52BG, 3;
	set $@kvm52BG_Victory, 2;

	stopnpctimer;
	sleep 1000;
	announce "- Red Team VENCEU! -",bc_all,0xff6347;
	sleep 3000;
	donpcevent "kvm52_BG::OnStop";
	end;

OnStop:
	if ($spectator_count10 > 0) {
       set $spectator_count10, $spectator_count10 - 1;
    }
	if ($spectator_count10 > 0) {
       set $spectator_count10, $spectator_count10 - 1;
    }
	// Warp Teams
	mapwarp "1@pump","izlude",128,105;
	donpcevent "kvm52_BG_Out::OnBegin";
	end;

OnReset:
	stopnpctimer;
	stopnpctimer "kvm52_BG_Out";
	set .Croix_Count, 0;
	set .Guillaume_Count, 0;
	set .Croix_Score, 0;
	set .Guillaume_Score, 0;
	set $@kvm52BG_Victory, 0;
	if( $@kvm52BG_id1 ) { bg_destroy $@kvm52BG_id1; set $@kvm52BG_id1, 0; }
	if( $@kvm52BG_id2 ) { bg_destroy $@kvm52BG_id2; set $@kvm52BG_id2, 0; }
	sleep 2000;
	mapwarp "1@pump","izlude",128,105;
	sleep 2000;
	maprespawnguildid "1@pump",0,3; // Just in case someone else
	bg_updatescore "1@pump",0,0;
	sleep 2000;
	set $@kvm52BG, 0;
	donpcevent "kvm52_BG::OnReadyCheck"; // Maybe a game is ready to start
	end;
}

-	script	kvm52_BG_Out	-1,{
	end;

OnBegin:
	initnpctimer;
	end;

OnTimer50000:
	mapannounce "1@pump","- The match will end in 10 seconds! -",1,0x696969;
	end;
OnTimer3000:
	donpcevent "kvm52_BG::OnReset";
	end;
}
