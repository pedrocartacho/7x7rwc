// Battleground Engine
// *********************************************************************

-	script	KvM39_BG	-1,{
	end;

OnGuillaumeJoin:
OnCroixJoin:
	donpcevent "KvM39_BG::OnReadyCheck";
	end;

OnGuillaumeQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnGuillaumeDie:
	if( $@KvM39BG == 2 )
	{
		set .Guillaume_Count, .Guillaume_Count - 1;
		set .Croix_Score, .Croix_Score + 1;
		bg_updatescore "2012rwc_01",.Guillaume_Score,.Croix_Score;
		if( .Guillaume_Count < 1 ){
			donpcevent "KvM39_BG::OnCroixWin";
		}
	}
	end;

OnCroixQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnCroixDie:
	if( $@KvM39BG == 2 )
	{
		set .Croix_Count, .Croix_Count - 1;
		set .Guillaume_Score, .Guillaume_Score + 1;
		bg_updatescore "2012rwc_01",.Guillaume_Score,.Croix_Score;
		if( .Croix_Count < 1 ){
			donpcevent "KvM39_BG::OnGuillaumeWin";
		}
	}
	end;

OnReadyCheck:
	if( $@KvM39BG )
		end;
	set .@Guillaume, getwaitingroomstate(0,"KvM39R_Guillaume");
	set .@Croix, getwaitingroomstate(0,"KvM39R_Croix");

	if((.@Guillaume < $@jogadoresblue) || (.@Croix < $@jogadoresred) )
	{
		if( .@Guillaume > 3 && .@Croix > 3 && !agitcheck() && $@KvM_Flood < gettimetick(2) )
		{
			announce "Arena 9 - 2012rwc_01 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",0,0xFFA500;
			set $@KvM_Flood, gettimetick(2) + 15;
		}
		else
			mapannounce "quiz_02","Arena 9 - 2012rwc_01 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",1,0xFFA500;
		end;
	}
		
	if ((.@Guillaume == 0) || (.@Croix == 0)) {
		end;
	}
	
	set $@KvM39BG, 1; // Starting
	donpcevent "KvM39R_Croix::OnEnterBG";
	donpcevent "KvM39R_Guillaume::OnEnterBG";
	donpcevent "KvM39_BG::OnStart";
	end;

OnStart:
	announce "- Duas equipes entraram na [ Arena 9 - 2012rwc_01 ]. -", bc_yellow;
	set $@KvM39BG_Victory, 0;
	// Warp Teams
	bg_warp $@KvM39BG_id1,"2012rwc_01",102,110;
	bg_warp $@KvM39BG_id2,"2012rwc_01",241,109;

	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }
	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }

	setwall "2012rwc_01",114,113,8,4,0,"batc01wall_a";
	setwall "2012rwc_01",228,113,8,4,0,"batc01wall_d";
	setwall "2012rwc_01",229,113,8,4,0,"batc01wall_ae01";
	setwall "2012rwc_01",113,113,8,4,0,"batc01wall_ae02";

	sleep 100;
	// ScoreBoard
	
	set .Guillaume_Score, 0;
	set .Croix_Score, 0;
	initnpctimer;
	bg_updatescore "2012rwc_01",.Guillaume_Score,.Croix_Score;
	end;

OnTimer3000:
	mapannounce "2012rwc_01","- The match will begin in 15 seconds! -", bc_yellow;
	end;

OnTimer15000:
	mapannounce "2012rwc_01","- The match will begin in 5 seconds! -", bc_yellow;
	end;

OnTimer16000:
	mapannounce "2012rwc_01","- The match will begin in 4 seconds! -", bc_yellow;
	end;

OnTimer17000:
	mapannounce "2012rwc_01","- The match will begin in 3 seconds! -", bc_yellow;
	end;

OnTimer18000:
	mapannounce "2012rwc_01","- The match will begin in 2 seconds! -", bc_yellow;
	end;
	
OnTimer19000:
	mapannounce "2012rwc_01","- The match will begin in 1 second! -", bc_yellow;
	end;

OnTimer20000:
	// Team Members
	set .Guillaume_Count, bg_get_data($@KvM39BG_id1, 0);
	set .Croix_Count, bg_get_data($@KvM39BG_id2, 0);
	if( (.Guillaume_Count < $@jogadoresblue) || (.Croix_Count < $@jogadoresred) )
	{
		set $@KvM39BG_Victory, 3;
		set $@KvM39BG, 3;
		mapannounce "2012rwc_01","- There are not enough players to start the match. -", bc_yellow;
		stopnpctimer;
		sleep 2000;
		donpcevent "KvM39_BG::OnStop";
		end;
	}

	set $@KvM39BG, 2; // Playing
	delwall "batc01wall_a";
	delwall "batc01wall_d";
	delwall "batc01wall_ae01";
	delwall "batc01wall_ae02";
	end;

OnTimer25001:
	mapannounce "2012rwc_01","- The match started! Good luck to all. -", bc_yellow;
killmonsterall "2012rwc_01";
	end;

OnTimer420000:
	mapannounce "2012rwc_01","- The match will end in 30 seconds! -", bc_yellow;
	end;

OnTimer450000:
	if( .Croix_Count > .Guillaume_Count )
		donpcevent "KvM39_BG::OnCroixWin";
	else if( .Croix_Count < .Guillaume_Count )
		donpcevent "KvM39_BG::OnGuillaumeWin";
	else
	{ // Draw Game
		set $@KvM39BG, 3;
		set $@KvM39BG_Victory, 3;

		stopnpctimer;
		sleep 2000;
		announce "- As duas equipes empataram! -", bc_yellow;
		donpcevent "KvM39_BG::OnStop";
	}
	end;

OnGuillaumeWin:
	set $@KvM39BG, 3;
	set $@KvM39BG_Victory, 1;

	stopnpctimer;
	sleep 1000;	
	announce "- Blue Team VENCEU! -",bc_all,0x00bfff;
	sleep 3000;
	donpcevent "KvM39_BG::OnStop";
	end;

OnCroixWin:
	set $@KvM39BG, 3;
	set $@KvM39BG_Victory, 2;

	stopnpctimer;
	sleep 1000;
	announce "- Red Team VENCEU! -",bc_all,0xff6347;
	sleep 3000;
	donpcevent "KvM39_BG::OnStop";
	end;

OnStop:
	if ($spectator_count9 > 0) {
       set $spectator_count9, $spectator_count9 - 1;
    }
	if ($spectator_count9 > 0) {
       set $spectator_count9, $spectator_count9 - 1;
    }
	// Warp Teams
	mapwarp "2012rwc_01","izlude",128,105;
	donpcevent "KvM39_BG_Out::OnBegin";
	end;

OnReset:
	stopnpctimer;
	stopnpctimer "KvM39_BG_Out";
	set .Croix_Count, 0;
	set .Guillaume_Count, 0;
	set .Croix_Score, 0;
	set .Guillaume_Score, 0;
	set $@KvM39BG_Victory, 0;
	if( $@KvM39BG_id1 ) { bg_destroy $@KvM39BG_id1; set $@KvM39BG_id1, 0; }
	if( $@KvM39BG_id2 ) { bg_destroy $@KvM39BG_id2; set $@KvM39BG_id2, 0; }
	sleep 2000;
	mapwarp "2012rwc_01","izlude",128,105;
	sleep 2000;
	maprespawnguildid "2012rwc_01",0,3; // Just in case someone else
	bg_updatescore "2012rwc_01",0,0;
	sleep 2000;
	set $@KvM39BG, 0;
	donpcevent "KvM39_BG::OnReadyCheck"; // Maybe a game is ready to start
	end;
}

-	script	KvM39_BG_Out	-1,{
	end;

OnBegin:
	initnpctimer;
	end;

OnTimer50000:
	mapannounce "2012rwc_01","- The match will end in 10 seconds! -",1,0x696969;
	end;
OnTimer3000:
	donpcevent "KvM39_BG::OnReset";
	end;
}
