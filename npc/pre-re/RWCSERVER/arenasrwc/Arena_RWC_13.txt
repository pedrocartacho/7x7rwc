// Battleground Engine
// *********************************************************************

-	script	kvm43_BG	-1,{
	end;

OnGuillaumeJoin:
OnCroixJoin:
	donpcevent "kvm43_BG::OnReadyCheck";
	end;

OnGuillaumeQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnGuillaumeDie:
	if( $@kvm43BG == 2 )
	{
		set .Guillaume_Count, .Guillaume_Count - 1;
		set .Croix_Score, .Croix_Score + 1;
		bg_updatescore "rwc01",.Guillaume_Score,.Croix_Score;
		if( .Guillaume_Count < 1 ){
			donpcevent "kvm43_BG::OnCroixWin";
		}
	}
	end;

OnCroixQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnCroixDie:
	if( $@kvm43BG == 2 )
	{
		set .Croix_Count, .Croix_Count - 1;
		set .Guillaume_Score, .Guillaume_Score + 1;
		bg_updatescore "rwc01",.Guillaume_Score,.Croix_Score;
		if( .Croix_Count < 1 ){
			donpcevent "kvm43_BG::OnGuillaumeWin";
		}
	}
	end;

OnReadyCheck:
	if( $@kvm43BG )
		end;
	set .@Guillaume, getwaitingroomstate(0,"kvm43R_Guillaume");
	set .@Croix, getwaitingroomstate(0,"kvm43R_Croix");

	if((.@Guillaume < $@jogadoresblue) || (.@Croix < $@jogadoresred) )
	{
		if( .@Guillaume > 3 && .@Croix > 3 && !agitcheck() && $@KvM_Flood < gettimetick(2) )
		{
			announce "Arena 13 - rwc01 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",0,0xFFA500;
			set $@KvM_Flood, gettimetick(2) + 15;
		}
		else
			mapannounce "quiz_02","Arena 13 - rwc01 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",1,0xFFA500;
		end;
	}
		
	if ((.@Guillaume == 0) || (.@Croix == 0)) {
		end;
	}
	
	set $@kvm43BG, 1; // Starting
	donpcevent "kvm43R_Croix::OnEnterBG";
	donpcevent "kvm43R_Guillaume::OnEnterBG";
	donpcevent "kvm43_BG::OnStart";
	end;

OnStart:
	announce "- Duas equipes entraram na [ Arena 13 - rwc01 ]. -", bc_yellow;
	set $@kvm43BG_Victory, 0;
	// Warp Teams
	bg_warp $@kvm43BG_id1,"rwc01",10,50;
	bg_warp $@kvm43BG_id2,"rwc01",89,50;

	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }
	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }

	setwall "rwc01",17,53,8,4,0,"batc01wall_a";
	setwall "rwc01",82,53,8,4,0,"batc01wall_d";
	setwall "rwc01",18,53,8,4,0,"batc01wall_ae01";
	setwall "rwc01",83,53,8,4,0,"batc01wall_ae02";

	sleep 100;
	// ScoreBoard
	
	set .Guillaume_Score, 0;
	set .Croix_Score, 0;
	initnpctimer;
	bg_updatescore "rwc01",.Guillaume_Score,.Croix_Score;
	end;

OnTimer3000:
	mapannounce "rwc01","- The match will begin in 15 seconds! -", bc_yellow;
	end;

OnTimer15000:
	mapannounce "rwc01","- The match will begin in 5 seconds! -", bc_yellow;
	end;

OnTimer16000:
	mapannounce "rwc01","- The match will begin in 4 seconds! -", bc_yellow;
	end;

OnTimer17000:
	mapannounce "rwc01","- The match will begin in 3 seconds! -", bc_yellow;
	end;

OnTimer18000:
	mapannounce "rwc01","- The match will begin in 2 seconds! -", bc_yellow;
	end;
	
OnTimer19000:
	mapannounce "rwc01","- The match will begin in 1 second! -", bc_yellow;
	end;

OnTimer20000:
	// Team Members
	set .Guillaume_Count, bg_get_data($@kvm43BG_id1, 0);
	set .Croix_Count, bg_get_data($@kvm43BG_id2, 0);
	if( (.Guillaume_Count < $@jogadoresblue) || (.Croix_Count < $@jogadoresred) )
	{
		set $@kvm43BG_Victory, 3;
		set $@kvm43BG, 3;
		mapannounce "rwc01","- There are not enough players to start the match. -", bc_yellow;
		stopnpctimer;
		sleep 2000;
		donpcevent "kvm43_BG::OnStop";
		end;
	}

	set $@kvm43BG, 2; // Playing
	delwall "batc01wall_a";
	delwall "batc01wall_d";
	delwall "batc01wall_ae01";
	delwall "batc01wall_ae02";
	end;

OnTimer25001:
	mapannounce "rwc01","- The match started! Good luck to all. -", bc_yellow;
killmonsterall "rwc01";
	end;

OnTimer420000:
	mapannounce "rwc01","- The match will end in 30 seconds! -", bc_yellow;
	end;

OnTimer450000:
	if( .Croix_Count > .Guillaume_Count )
		donpcevent "kvm43_BG::OnCroixWin";
	else if( .Croix_Count < .Guillaume_Count )
		donpcevent "kvm43_BG::OnGuillaumeWin";
	else
	{ // Draw Game
		set $@kvm43BG, 3;
		set $@kvm43BG_Victory, 3;

		stopnpctimer;
		sleep 2000;
		announce "- As duas equipes empataram! -", bc_yellow;
		donpcevent "kvm43_BG::OnStop";
	}
	end;

OnGuillaumeWin:
	set $@kvm43BG, 3;
	set $@kvm43BG_Victory, 1;

	stopnpctimer;
	sleep 1000;	
	announce "- Blue Team VENCEU! -",bc_all,0x00bfff;
	sleep 3000;
	donpcevent "kvm43_BG::OnStop";
	end;

OnCroixWin:
	set $@kvm43BG, 3;
	set $@kvm43BG_Victory, 2;

	stopnpctimer;
	sleep 1000;
	announce "- Red Team VENCEU! -",bc_all,0xff6347;
	sleep 3000;
	donpcevent "kvm43_BG::OnStop";
	end;

OnStop:
	if ($spectator_count9 > 0) {
       set $spectator_count9, $spectator_count9 - 1;
    }
	if ($spectator_count9 > 0) {
       set $spectator_count9, $spectator_count9 - 1;
    }
	// Warp Teams
	mapwarp "rwc01","izlude",128,105;
	donpcevent "kvm43_BG_Out::OnBegin";
	end;

OnReset:
	stopnpctimer;
	stopnpctimer "kvm43_BG_Out";
	set .Croix_Count, 0;
	set .Guillaume_Count, 0;
	set .Croix_Score, 0;
	set .Guillaume_Score, 0;
	set $@kvm43BG_Victory, 0;
	if( $@kvm43BG_id1 ) { bg_destroy $@kvm43BG_id1; set $@kvm43BG_id1, 0; }
	if( $@kvm43BG_id2 ) { bg_destroy $@kvm43BG_id2; set $@kvm43BG_id2, 0; }
	sleep 2000;
	mapwarp "rwc01","izlude",128,105;
	sleep 2000;
	maprespawnguildid "rwc01",0,3; // Just in case someone else
	bg_updatescore "rwc01",0,0;
	sleep 2000;
	set $@kvm43BG, 0;
	donpcevent "kvm43_BG::OnReadyCheck"; // Maybe a game is ready to start
	end;
}

-	script	kvm43_BG_Out	-1,{
	end;

OnBegin:
	initnpctimer;
	end;

OnTimer50000:
	mapannounce "rwc01","- The match will end in 10 seconds! -",1,0x696969;
	end;
OnTimer3000:
	donpcevent "kvm43_BG::OnReset";
	end;
}
