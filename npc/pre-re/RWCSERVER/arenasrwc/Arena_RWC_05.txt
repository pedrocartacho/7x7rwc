		if (getwaitingroomstate (1) != $@jogadoresbluechat){
			delwaitingroom;
			waitingroom "Arena 5 - 2009rwc_01",$@jogadoresbluechat,"KvM35_BG::OnGuillaumeJoin",1;
			}
	}
	end;
	
OnEnterBG:
	set $@KvM35BG_id1, waitingroom2bg("izlude",128,105,"KvM35_BG::OnGuillaumeQuit","KvM35_BG::OnGuillaumeDie");

}

quiz_02,164,353,4	script	Waiting Room::KvM35R_Croix	414,{
	end;

OnInit:
	while (1){
		sleep 2000;
		if (getwaitingroomstate (1) != $@jogadoresredchat){
			delwaitingroom;
			waitingroom "Arena 5 - 2009rwc_01",$@jogadoresredchat,"KvM35_BG::OnCroixJoin",1;
			}
	}
	end;

OnEnterBG:
	set $@KvM35BG_id2, waitingroom2bg("izlude",128,105,"KvM35_BG::OnCroixQuit","KvM35_BG::OnCroixDie");
	end;
}

// Battleground Engine
// *********************************************************************

-	script	KvM35_BG	-1,{
	end;

OnGuillaumeJoin:
OnCroixJoin:
	donpcevent "KvM35_BG::OnReadyCheck";
	end;

OnGuillaumeQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnGuillaumeDie:
	if( $@KvM35BG == 2 )
	{
		set .Guillaume_Count, .Guillaume_Count - 1;
		set .Croix_Score, .Croix_Score + 1;
		bg_updatescore "x",.Guillaume_Score,.Croix_Score;
		if( .Guillaume_Count < 1 ){
			donpcevent "KvM35_BG::OnCroixWin";
		}
	}
	end;

OnCroixQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnCroixDie:
	if( $@KvM35BG == 2 )
	{
		set .Croix_Count, .Croix_Count - 1;
		set .Guillaume_Score, .Guillaume_Score + 1;
		bg_updatescore "2009rwc_01",.Guillaume_Score,.Croix_Score;
		if( .Croix_Count < 1 ){
			donpcevent "KvM35_BG::OnGuillaumeWin";
		}
	}
	end;

OnReadyCheck:
	if( $@KvM35BG )
		end;
	set .@Guillaume, getwaitingroomstate(0,"KvM35R_Guillaume");
	set .@Croix, getwaitingroomstate(0,"KvM35R_Croix");

	if((.@Guillaume < $@jogadoresblue) || (.@Croix < $@jogadoresred) )
	{
		if( .@Guillaume > 3 && .@Croix > 3 && !agitcheck() && $@KvM_Flood < gettimetick(2) )
		{
			announce "Arena 5 - 2009rwc_01 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",0,0xFFA500;
			set $@KvM_Flood, gettimetick(2) + 15;
		}
		else
			mapannounce "quiz_02","Arena 5 - 2009rwc_01: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",1,0xFFA500;
		end;
	}
		
	if ((.@Guillaume == 0) || (.@Croix == 0)) {
		end;
	}
	
	set $@KvM35BG, 1; // Starting
	donpcevent "KvM35R_Croix::OnEnterBG";
	donpcevent "KvM35R_Guillaume::OnEnterBG";
	donpcevent "KvM35_BG::OnStart";
	end;

OnStart:
	announce "- Duas equipes entraram na [ Arena 5 - 2009rwc_01 ]. -", bc_yellow;
	set $@KvM35BG_Victory, 0;
	// Warp Teams
	bg_warp $@KvM35BG_id1,"2009rwc_01",19,50;
	bg_warp $@KvM35BG_id2,"2009rwc_01",80,50;

	if ($@guildsInArena > 0) {
        $@guildsInArena--;
    }
	if ($@guildsInArena > 0) {
        $@guildsInArena--;
    }

	setwall "2009rwc_01",25,55,2,4,0,"9a";
	setwall "2009rwc_01",25,51,4,4,0,"9b";
	setwall "2009rwc_01",25,45,2,4,0,"9c";
	setwall "2009rwc_01",74,55,2,4,0,"9d";
	setwall "2009rwc_01",74,51,4,4,0,"9e";
	setwall "2009rwc_01",74,45,2,4,0,"9f";


	sleep 100;
	// ScoreBoard
	
	set .Guillaume_Score, 0;
	set .Croix_Score, 0;
	initnpctimer;
	bg_updatescore "2009rwc_01",.Guillaume_Score,.Croix_Score;
	end;

OnTimer3000:
	mapannounce "2009rwc_01","- The match will begin in 15 seconds! -", bc_yellow;
	end;

OnTimer15000:
	mapannounce "2009rwc_01","- The match will begin in 5 seconds! -", bc_yellow;
	end;

OnTimer16000:
	mapannounce "2009rwc_01","- The match will begin in 4 seconds! -", bc_yellow;
	end;

OnTimer17000:
	mapannounce "2009rwc_01","- The match will begin in 3 seconds! -", bc_yellow;
	end;

OnTimer18000:
	mapannounce "2009rwc_01","- The match will begin in 2 seconds! -", bc_yellow;
	end;
	
OnTimer19000:
	mapannounce "2009rwc_01","- The match will begin in 1 second! -", bc_yellow;
	end;

OnTimer20000:
	// Team Members
	set .Guillaume_Count, bg_get_data($@KvM35BG_id1, 0);
	set .Croix_Count, bg_get_data($@KvM35BG_id2, 0);
	if( (.Guillaume_Count < $@jogadoresblue) || (.Croix_Count < $@jogadoresred) )
	{
		set $@KvM35BG_Victory, 3;
		set $@KvM35BG, 3;
		mapannounce "2009rwc_01","- There are not enough players to start the match. -", bc_yellow;
		stopnpctimer;
		sleep 2000;
		donpcevent "KvM35_BG::OnStop";
		end;
	}

	set $@KvM35BG, 2; // Playing
	delwall "9a";
	delwall "9b";
	delwall "9c";
	delwall "9d";
	delwall "9e";
	delwall "9f";
	end;

OnTimer25001:
	mapannounce "2009rwc_01","- The match started! Good luck to all. -", bc_yellow;
killmonsterall "2009rwc_01";
	end;

OnTimer420000:
	mapannounce "2009rwc_01","- The match will end in 30 seconds! -", bc_yellow;
	end;

OnTimer450000:
	if( .Croix_Count > .Guillaume_Count )
		donpcevent "KvM35_BG::OnCroixWin";
	else if( .Croix_Count < .Guillaume_Count )
		donpcevent "KvM35_BG::OnGuillaumeWin";
	else
	{ // Draw Game
		set $@KvM35BG, 3;
		set $@KvM35BG_Victory, 3;

		stopnpctimer;
		sleep 2000;
		announce "- As duas equipes empataram! -", bc_yellow;
		donpcevent "KvM35_BG::OnStop";
	}
	end;

OnGuillaumeWin:
	set $@KvM35BG, 3;
	set $@KvM35BG_Victory, 1;

	stopnpctimer;
	sleep 1000;	
	announce "- Blue Team VENCEU! -",bc_all,0x00bfff;
	sleep 3000;
	donpcevent "KvM35_BG::OnStop";
	end;

OnCroixWin:
	set $@KvM35BG, 3;
	set $@KvM35BG_Victory, 2;

	stopnpctimer;
	sleep 1000;
	announce "- Red Team VENCEU! -",bc_all,0xff6347;
	sleep 3000;
	donpcevent "KvM35_BG::OnStop";
	end;

OnStop:
	if ($spectator_count5 > 0) {
       set $spectator_count5, $spectator_count5 - 1;
    }
	if ($spectator_count5 > 0) {
       set $spectator_count5, $spectator_count5 - 1;
    }
	// Warp Teams
	mapwarp "2009rwc_01","izlude",128,105;
	donpcevent "KvM35_BG_Out::OnBegin";
	end;

OnReset:
	stopnpctimer;
	stopnpctimer "KvM35_BG_Out";
	set .Croix_Count, 0;
	set .Guillaume_Count, 0;
	set .Croix_Score, 0;
	set .Guillaume_Score, 0;
	set $@KvM35BG_Victory, 0;
	if( $@KvM35BG_id1 ) { bg_destroy $@KvM35BG_id1; set $@KvM35BG_id1, 0; }
	if( $@KvM35BG_id2 ) { bg_destroy $@KvM35BG_id2; set $@KvM35BG_id2, 0; }
	sleep 2000;
	mapwarp "2009rwc_01","izlude",128,105;
	sleep 2000;
	maprespawnguildid "2009rwc_01",0,3; // Just in case someone else
	bg_updatescore "2009rwc_01",0,0;
	sleep 2000;
	set $@KvM35BG, 0;
	donpcevent "KvM35_BG::OnReadyCheck"; // Maybe a game is ready to start
	end;
}

-	script	KvM35_BG_Out	-1,{
	end;

OnBegin:
	initnpctimer;
	end;

OnTimer50000:
	mapannounce "2009rwc_01","- The match will end in 10 seconds! -",1,0x696969;
	end;
OnTimer3000:
	donpcevent "KvM35_BG::OnReset";
	end;
}
