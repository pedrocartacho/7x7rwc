// Battleground Engine
// *********************************************************************

-	script	kvm49_BG	-1,{
	end;

OnGuillaumeJoin:
OnCroixJoin:
	donpcevent "kvm49_BG::OnReadyCheck";
	end;

OnGuillaumeQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnGuillaumeDie:
	if( $@kvm49BG == 2 )
	{
		set .Guillaume_Count, .Guillaume_Count - 1;
		set .Croix_Score, .Croix_Score + 1;
		bg_updatescore "1@dth3",.Guillaume_Score,.Croix_Score;
		if( .Guillaume_Count < 1 ){
			donpcevent "kvm49_BG::OnCroixWin";
		}
	}
	end;

OnCroixQuit:
	set BG_Delay_Tick, gettimetick(2) + 2000;
OnCroixDie:
	if( $@kvm49BG == 2 )
	{
		set .Croix_Count, .Croix_Count - 1;
		set .Guillaume_Score, .Guillaume_Score + 1;
		bg_updatescore "1@dth3",.Guillaume_Score,.Croix_Score;
		if( .Croix_Count < 1 ){
			donpcevent "kvm49_BG::OnGuillaumeWin";
		}
	}
	end;

OnReadyCheck:
	if( $@kvm49BG )
		end;
	set .@Guillaume, getwaitingroomstate(0,"kvm49R_Guillaume");
	set .@Croix, getwaitingroomstate(0,"kvm49R_Croix");

	if((.@Guillaume < $@jogadoresblue) || (.@Croix < $@jogadoresred))
	{
		if( .@Guillaume > 1 && .@Croix > 1 && !agitcheck() && $@KvM_Flood < gettimetick(2) )
		{
			announce "Arena 19 - 1@dth3 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",0,0x00bfff;
			set $@KvM_Flood, gettimetick(2) + 15;
		}
		else
			mapannounce "quiz_02","Arena 19 - 1@dth3 | Blue Team: " + .@Guillaume + "/"+$@jogadoresblue+", Red Team: " + .@Croix + "/"+$@jogadoresred+"",1,0x00bfff;
		end;
	}

	if ((.@Guillaume == 0) || (.@Croix == 0)) {
		end;
	}
	
	set $@kvm49BG, 1; // Starting
	donpcevent "kvm49R_Croix::OnEnterBG";
	donpcevent "kvm49R_Guillaume::OnEnterBG";
	donpcevent "kvm49_BG::OnStart";
	end;

OnStart:
	announce "- Duas equipes entraram na [ Arena 19 - 1@dth3 ]. -", bc_yellow;
	set $@kvm49BG_Victory, 0;
	// Warp Teams
	bg_warp $@kvm49BG_id1,"1@dth3",52,89;
	bg_warp $@kvm49BG_id2,"1@dth3",88,48;

	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }
	if ($@guildsInArena2 > 0) {
        $@guildsInArena2--;
    }

	setwall "1@dth3",48,93,10,6,0,"batc20wall_a";
	setwall "1@dth3",48,92,9,4,0,"batc20wall_b";
	setwall "1@dth3",57,84,9,2,0,"batc20wall_c";
	setwall "1@dth3",57,85,8,0,0,"batc20wall_d";
	setwall "1@dth3",83,53,10,6,0,"batc20wall_e";
	setwall "1@dth3",83,52,9,4,0,"batc20wall_f";
	setwall "1@dth3",92,44,9,2,0,"batc20wall_g";
	setwall "1@dth3",92,45,8,0,0,"batc20wall_h";

	sleep 100;
	// ScoreBoard
	
	set .Guillaume_Score, 0;
	set .Croix_Score, 0;
	initnpctimer;
	end;

OnTimer3000:
	mapannounce "1@dth3","- The match will begin in 15 seconds! -", bc_yellow;
	end;

OnTimer15000:
	mapannounce "1@dth3","- The match will begin in 5 seconds! -", bc_yellow;
	end;

OnTimer16000:
	mapannounce "1@dth3","- The match will begin in 4 seconds! -", bc_yellow;
	end;

OnTimer17000:
	mapannounce "1@dth3","- The match will begin in 3 seconds! -", bc_yellow;
	end;

OnTimer18000:
	mapannounce "1@dth3","- The match will begin in 2 seconds! -", bc_yellow;
	end;
	
OnTimer19000:
	mapannounce "1@dth3","- The match will begin in 1 second! -", bc_yellow;
	end;

OnTimer20000:
	// Team Members
	set .Guillaume_Count, bg_get_data($@kvm49BG_id1, 0);
	set .Croix_Count, bg_get_data($@kvm49BG_id2, 0);
	if( (.Guillaume_Count < $@jogadoresblue) || (.Croix_Count < $@jogadoresred) )
	{
		set $@kvm49BG_Victory, 3;
		set $@kvm49BG, 3;
		mapannounce "1@dth3","- There are not enough players to start the match. -", bc_yellow;
		stopnpctimer;
		sleep 2000;
		donpcevent "kvm49_BG::OnStop";
		end;
	}

	set $@kvm49BG, 2; // Playing
	delwall "batc20wall_a";
	delwall "batc20wall_b";
	delwall "batc20wall_c";
	delwall "batc20wall_d";
	end;

OnTimer25001:
	mapannounce "1@dth3","- The match started! Good luck to all. -", bc_yellow;
killmonsterall "1@dth3";
	end;

OnTimer420000:
	mapannounce "1@dth3","- The match will end in 30 seconds! -", bc_yellow;
	end;

OnTimer450000:
	if( .Croix_Count > .Guillaume_Count )
		donpcevent "kvm49_BG::OnCroixWin";
	else if( .Croix_Count < .Guillaume_Count )
		donpcevent "kvm49_BG::OnGuillaumeWin";
	else
	{ // Draw Game
		set $@kvm49BG, 3;
		set $@kvm49BG_Victory, 3;

		stopnpctimer;
		sleep 2000;
		announce "- As duas equipes empataram! -", bc_yellow;
		donpcevent "kvm49_BG::OnStop";
	}
	end;

OnGuillaumeWin:
	set $@kvm49BG, 3;
	set $@kvm49BG_Victory, 1;

	stopnpctimer;
	sleep 1000;	
	announce "- Blue Team VENCEU! -",bc_all,0x00bfff;
	sleep 3000;
	donpcevent "kvm49_BG::OnStop";
	end;

OnCroixWin:
	set $@kvm49BG, 3;
	set $@kvm49BG_Victory, 2;

	stopnpctimer;
	sleep 1000;
	announce "- Red Team VENCEU! -",bc_all,0xff6347;
	sleep 3000;
	donpcevent "kvm49_BG::OnStop";
	end;

OnStop:
	if ($spectator_count10 > 0) {
       set $spectator_count10, $spectator_count10 - 1;
    }
	if ($spectator_count10 > 0) {
       set $spectator_count10, $spectator_count10 - 1;
    }
	// Warp Teams
	mapwarp "1@dth3","izlude",128,105;
	donpcevent "kvm49_BG_Out::OnBegin";
	end;

OnReset:
	stopnpctimer;
	stopnpctimer "kvm49_BG_Out";
	set .Croix_Count, 0;
	set .Guillaume_Count, 0;
	set .Croix_Score, 0;
	set .Guillaume_Score, 0;
	set $@kvm49BG_Victory, 0;
	if( $@kvm49BG_id1 ) { bg_destroy $@kvm49BG_id1; set $@kvm49BG_id1, 0; }
	if( $@kvm49BG_id2 ) { bg_destroy $@kvm49BG_id2; set $@kvm49BG_id2, 0; }
	sleep 2000;
	mapwarp "1@dth3","izlude",128,105;
	sleep 2000;
	maprespawnguildid "1@dth3",0,3; // Just in case someone else
	bg_updatescore "1@dth3",0,0;
	sleep 2000;
	set $@kvm49BG, 0;
	donpcevent "kvm49_BG::OnReadyCheck"; // Maybe a game is ready to start
	end;
}

-	script	kvm49_BG_Out	-1,{
	end;

OnBegin:
	initnpctimer;
	end;

OnTimer50000:
	mapannounce "1@dth3","- The match will end in 10 seconds! -",1,0x696969;
	end;
OnTimer3000:
	donpcevent "kvm49_BG::OnReset";
	end;
}
